```python
from groq import Groq

critic_client = Groq(api_key=os.getenv('GROQ_API_KEY'))

def call_critic(prompt: str) -> dict:
    response = critic_client.chat.completions.create(
        model="llama-3.1-70b-versatile",
        messages=[
            {"role": "system", "content": CRITIC_SYSTEM_PROMPT},
            {"role": "user", "content": prompt}
        ],
        temperature=0.2,  # Lower temperature for more deterministic validation
        max_tokens=1000,
        response_format={"type": "json_object"}
    )
    return json.loads(response.choices[0].message.content)
```

### 1.2 Other Implementation

#### **1.2.1 Orchestration: LangGraph Workflow**

**State Definition:**
```python
from typing import TypedDict, List, Optional
from langgraph.graph import StateGraph, END

class AgentState(TypedDict):
    task_id: str
    original_prompt: str
    constraints: List[str]
    current_step: int
    generated_steps: List[dict]  # [{code, reasoning, validation_status}]
    drift_detected: bool
    correction_attempts: int
    final_code: Optional[str]
    session_id: int
```

**Workflow Nodes:**
```python
def generate_code_step(state: AgentState) -> AgentState:
    """Generate next code step using Gemini."""
    step_num = state['current_step']
    
    # Build context from previous steps
    context = {
        "task": state['original_prompt'],
        "constraints": state['constraints'],
        "previous_steps": state['generated_steps']
    }
    
    prompt = f"""
    Task: {context['task']}
    Constraints: {context['constraints']}
    Previous steps completed: {len(context['previous_steps'])}
    
    Generate step {step_num}. Context from previous steps:
    {json.dumps(context['previous_steps'], indent=2)}
    """
    
    response = generator_model.generate_content(prompt)
    step_data = json.loads(response.text)
    
    # Save to database
    save_step_to_db(state['session_id'], step_data)
    
    state['generated_steps'].append(step_data)
    return state

def validate_step(state: AgentState) -> AgentState:
    """Validate current step using AST + LLM Critic."""
    current_step = state['generated_steps'][-1]
    
    # 1. AST Syntax Check
    ast_valid = validate_ast(current_step['code'])
    if not ast_valid['valid']:
        state['drift_detected'] = True
        log_validation(state['session_id'], 'ast', False, ast_valid['message'])
        return state
    
    # 2. Rule-based checks
    rule_violations = check_drift_rules(
        current_step['code'], 
        state['constraints'],
        state['generated_steps'][:-1]
    )
    if rule_violations:
        state['drift_detected'] = True
        log_validation(state['session_id'], 'rules', False, rule_violations)
        return state
    
    # 3. LLM Critic Evaluation
    critic_prompt = f"""
    Original Task: {state['original_prompt']}
    Constraints: {state['constraints']}
    
    Previous Steps:
    {json.dumps(state['generated_steps'][:-1], indent=2)}
    
    Current Step to Evaluate:
    Code: {current_step['code']}
    Reasoning: {current_step['reasoning']}
    
    Question: Does this step maintain logical consistency?
    """
    
    critic_result = call_critic(critic_prompt)
    log_validation(state['session_id'], 'llm_critic', 
                   not critic_result['drift_detected'], 
                   critic_result['explanation'])
    
    state['drift_detected'] = critic_result['drift_detected']
    if critic_result['drift_detected']:
        state['generated_steps'][-1]['drift_info'] = critic_result
    
    return state

def regenerate_step(state: AgentState) -> AgentState:
    """Regenerate faulty step with corrective prompt."""
    faulty_step = state['generated_steps'][-1]
    drift_info = faulty_step.get('drift_info', {})
    
    corrective_prompt = f"""
    CORRECTION NEEDED:
    
    Original Task: {state['original_prompt']}
    
    Your previous attempt for step {state['current_step']} had this issue:
    {drift_info.get('explanation', 'Validation failed')}
    
    Previous valid steps (DO NOT MODIFY):
    {json.dumps(state['generated_steps'][:-1], indent=2)}
    
    Task: Regenerate ONLY step {state['current_step']} to fix:
    {drift_info.get('suggestion', 'the identified issue')}
    
    Ensure this step:
    1. Addresses: {faulty_step.get('addresses_requirement', 'the stated requirement')}
    2. Does NOT contradict previous steps
    3. Adheres to all constraints: {state['constraints']}
    """
    
    response = generator_model.generate_content(corrective_prompt)
    corrected_step = json.loads(response.text)
    
    # Log regeneration attempt
    log_regeneration(state['session_id'], state['current_step'], 
                     state['correction_attempts'], corrective_prompt, 
                     corrected_step['code'])
    
    # Replace faulty step
    state['generated_steps'][-1] = corrected_step
    state['correction_attempts'] += 1
    
    return state

def should_regenerate(state: AgentState) -> str:
    """Decision function for conditional branching."""
    if state['drift_detected']:
        if state['correction_attempts'] < 3:  # Max 3 attempts
            return "regenerate"
        else:
            return "fail"  # Give up after 3 attempts
    return "continue"

def finalize_code(state: AgentState) -> AgentState:
    """Combine all steps into final code."""
    final_code = "\n\n".join([step['code'] for step in state['generated_steps']])
    state['final_code'] = final_code
    
    # Save to database
    save_final_code(state['session_id'], final_code)
    
    return state
```

**LangGraph Workflow Construction:**
```python
workflow = StateGraph(AgentState)

# Add nodes
workflow.add_node("generate", generate_code_step)
workflow.add_node("validate", validate_step)
workflow.add_node("regenerate", regenerate_step)
workflow.add_node("finalize", finalize_code)

# Define edges
workflow.set_entry_point("generate")
workflow.add_edge("generate", "validate")

# Conditional branching after validation
workflow.add_conditional_edges(
    "validate",
    should_regenerate,
    {
        "regenerate": "regenerate",
        "continue": "generate",  # Next step
        "fail": "finalize"
    }
)

# After regeneration, re-validate
workflow.add_edge("regenerate", "validate")

# Finalize leads to end
workflow.add_edge("finalize", END)

app = workflow.compile()
```

#### **1.2.2 Validation Layer**

**AST Syntax Validator:**
```python
import ast

def validate_ast(code: str) -> dict:
    """Deterministic syntax validation."""
    try:
        tree = ast.parse(code)
        return {
            "valid": True,
            "message": "Syntax valid",
            "ast_tree": tree
        }
    except SyntaxError as e:
        return {
            "valid": False,
            "message": f"Syntax error at line {e.lineno}: {e.msg}",
            "error_type": "SyntaxError"
        }
```

**Drift Rule Engine:**
```python
def check_drift_rules(current_code: str, constraints: List[str], 
                      previous_steps: List[dict]) -> Optional[str]:
    """Apply hardcoded drift detection rules."""
    
    # Rule 1: Signature drift
    if previous_steps:
        prev_signature = extract_function_signature(previous_steps[0]['code'])
        curr_signature = extract_function_signature(current_code)
        if prev_signature and curr_signature and prev_signature != curr_signature:
            return f"Signature drift: Changed from {prev_signature} to {curr_signature}"
    
    # Rule 2: Unauthorized imports
    current_imports = extract_imports(current_code)
    allowed_imports = extract_allowed_from_constraints(constraints)
    unauthorized = set(current_imports) - set(allowed_imports)
    if unauthorized:
        return f"Unauthorized imports: {unauthorized}"
    
    # Rule 3: Undefined variable usage
    if previous_steps:
        defined_vars = extract_defined_variables(
            "\n".join([s['code'] for s in previous_steps])
        )
        used_vars = extract_used_variables(current_code)
        undefined = set(used_vars) - set(defined_vars) - set(get_builtins())
        if undefined:
            return f"References undefined variables: {undefined}"
    
    return None  # No drift detected

def extract_function_signature(code: str) -> Optional[str]:
    """Extract function def line from code."""
    try:
        tree = ast.parse(code)
        for node in ast.walk(tree):
            if isinstance(node, ast.FunctionDef):
                args = [arg.arg for arg in node.args.args]
                return f"{node.name}({', '.join(args)})"
    except:
        return None

def extract_imports(code: str) -> List[str]:
    """Extract all imported modules."""
    try:
        tree = ast.parse(code)
        imports = []
        for node in ast.walk(tree):
            if isinstance(node, ast.Import):
                imports.extend([alias.name for alias in node.names])
            elif isinstance(node, ast.ImportFrom):
                imports.append(node.module)
        return imports
    except:
        return []
```

**Schema:**
```sql
-- Main task metadata
CREATE TABLE tasks (
    task_id TEXT PRIMARY KEY,
    prompt TEXT NOT NULL,
    function_signature TEXT,
    constraints TEXT,  -- JSON array
    test_suite TEXT,   -- JSON array of test cases
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

-- Generation sessions
CREATE TABLE generation_sessions (
    session_id INTEGER PRIMARY KEY AUTOINCREMENT,
    task_id TEXT NOT NULL,
    system_type TEXT NOT NULL,  -- 'baseline' or 'self_correcting'
    model_generator TEXT,
    model_critic TEXT,
    start_time DATETIME DEFAULT CURRENT_TIMESTAMP,
    end_time DATETIME,
    status TEXT,  -- 'in_progress', 'completed', 'failed'
    FOREIGN KEY (task_id) REFERENCES tasks(task_id)
);

-- Individual generation steps
CREATE TABLE generation_steps (
    step_id INTEGER PRIMARY KEY AUTOINCREMENT,
    session_id INTEGER NOT NULL,
    step_number INTEGER NOT NULL,
    code TEXT NOT NULL,
    reasoning TEXT,
    addresses_requirement TEXT,
    assumptions TEXT,  -- JSON array
    validation_status TEXT,  -- 'pending', 'valid', 'drifted'
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (session_id) REFERENCES generation_sessions(session_id)
);

-- Validation results
CREATE TABLE validations (
    validation_id INTEGER PRIMARY KEY AUTOINCREMENT,
    step_id INTEGER NOT NULL,
    validator_type TEXT NOT NULL,  -- 'ast', 'rules', 'llm_critic'
    passed BOOLEAN NOT NULL,
    confidence_score REAL,
    feedback TEXT,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (step_id) REFERENCES generation_steps(step_id)
);

-- Regeneration attempts
CREATE TABLE regenerations (
    regeneration_id INTEGER PRIMARY KEY AUTOINCREMENT,
    original_step_id INTEGER NOT NULL,
    attempt_number INTEGER NOT NULL,
    corrective_prompt TEXT,
    regenerated_code TEXT,
    succeeded BOOLEAN,

-- Final results

    result_id INTEGER PRIMARY KEY AUTOINCREMENT,
    session_id INTEGER NOT NULL,

    total_corrections INTEGER,
    execution_time_seconds REAL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,

);
```

**Database Helper Functions:**
```python
import sqlite3
from datetime import datetime

class StateDatabase:
    def __init__(self, db_path='agent_state.db'):
        self.conn = sqlite3.connect(db_path)
        self.cursor = self.conn.cursor()
    
    def create_session(self, task_id: str, system_type: str) -> int:
        """Create new generation session."""
        self.cursor.execute("""
            INSERT INTO generation_sessions (task_id, system_type, status)
            VALUES (?, ?, 'in_progress')
        """, (task_id, system_type))
        self.conn.commit()
        return self.cursor.lastrowid
    
    def save_step(self, session_id: int, step_data: dict) -> int:
        """Save generated step."""
        self.cursor.execute("""
            INSERT INTO generation_steps 
            (session_id, step_number, code, reasoning, addresses_requirement, assumptions)
            VALUES (?, ?, ?, ?, ?, ?)
        """, (
            session_id,
            step_data['step_number'],
            step_data['code'],
            step_data['reasoning'],
            step_data.get('addresses_requirement'),
            json.dumps(step_data.get('assumptions', []))
        ))
        self.conn.commit()
        return self.cursor.lastrowid
    
    def log_validation(self, step_id: int, validator_type: str, 
                      passed: bool, feedback: str):
        """Log validation result."""
        self.cursor.execute("""
            INSERT INTO validations (step_id, validator_type, passed, feedback)
            VALUES (?, ?, ?, ?)
        """, (step_id, validator_type, passed, feedback))
        self.conn.commit()
    
    def get_last_valid_state(self, session_id: int) -> List[dict]:
        """Retrieve all validated steps for backtracking."""
        self.cursor.execute("""
            SELECT code, reasoning FROM generation_steps
            WHERE session_id = ? AND validation_status = 'valid'
            ORDER BY step_number
        """, (session_id,))
        return [{"code": row[0], "reasoning": row[1]} for row in self.cursor.fetchall()]
```

```typescript
import * as vscode from 'vscode';
import { generateWithCorrection } from './api';

export function activate(context: vscode.ExtensionContext) {
    let disposable = vscode.commands.registerCommand(
        'self-correcting-agent.generate',
        async () => {
            const editor = vscode.window.activeTextEditor;
            if (!editor) {
                vscode.window.showErrorMessage('No active editor');
                return;
            }

            // Get user prompt
            const prompt = await vscode.window.showInputBox({
                prompt: 'Describe the code you want to generate',
                placeHolder: 'e.g., Write a function to calculate factorial'
            });

            if (!prompt) return;

            // Show progress
            await vscode.window.withProgress({
                location: vscode.ProgressLocation.Notification,
                title: 'Generating code with drift detection...',
                cancellable: false
            }, async (progress) => {
                try {
                    // Call backend API
                    const result = await generateWithCorrection(prompt);
                    
                    // Insert generated code
                    editor.edit(editBuilder => {
                        editBuilder.insert(
                            editor.selection.active,
                            result.final_code
                        );
                    });

                    // Show audit trail in output panel
                    const outputChannel = vscode.window.createOutputChannel(
                        'Self-Correcting Agent'
                    );
                    outputChannel.appendLine('=== Generation Report ===');
                    outputChannel.appendLine(`Steps: ${result.steps_count}`);
                    outputChannel.appendLine(`Corrections: ${result.corrections}`);
                    outputChannel.appendLine(`Tokens: ${result.tokens_used}`);
                    outputChannel.show();

                } catch (error) {
                    vscode.window.showErrorMessage(
                        `Generation failed: ${error.message}`
                    );
                }
            });
        }
    );

    context.subscriptions.push(disposable);
}
```

### 3.1 Results Quality - Success Metrics

**Primary Success Metric: Functional Correctness**

```python
def evaluate_functional_correctness(generated_code: str, test_suite: list) -> dict:
    """Execute HumanEval test cases against generated code."""
    
    passed_tests = 0
    failed_tests = []
    
    for test_case in test_suite:
        try:
            # Execute in isolated namespace
            namespace = {}
            exec(generated_code, namespace)
            exec(test_case['test_code'], namespace)
            passed_tests += 1
        except AssertionError as e:
            failed_tests.append({
                "test": test_case['test_code'],
                "error": str(e)
            })
        except Exception as e:
            failed_tests.append({
                "test": test_case['test_code'],
                "error": f"Runtime error: {str(e)}"
            })
    
    return {
        "passed": passed_tests,
        "total": len(test_suite),
        "pass_rate": passed_tests / len(test_suite),
        "failed_details": failed_tests
    }
```

**Success Definition:**
- **Pass@1:** Solution passes all test cases on first validated output
- **Pass@3:** Solution passes within 3 correction attempts
- **Functional Equivalence:** Generated code produces identical outputs to canonical solution

```python
def calculate_metrics(baseline_results: list, system_results: list) -> dict:
    """Calculate comparative metrics."""
    
    # Pass@1 Rate
    baseline_pass1 = sum(1 for r in baseline_results if r['passed']) / len(baseline_results)
    system_pass1 = sum(1 for r in system_results if r['passed']) / len(system_results)
    
    # Token Efficiency
    baseline_avg_tokens = sum(r['tokens'] for r in baseline_results) / len(baseline_results)
    system_avg_tokens = sum(r['tokens'] for r in system_results) / len(system_results)
    token_savings = ((baseline_avg_tokens - system_avg_tokens) / baseline_avg_tokens) * 100
    
    # Inference Latency
    baseline_avg_time = sum(r['time'] for r in baseline_results) / len(baseline_results)
    system_avg_time = sum(r['time'] for r in system_results) / len(system_results)
    
    # Correction Statistics
    avg_corrections = sum(r['correction_count'] for r in system_results) / len(system_results)
    correction_success_rate = sum(
        1 for r in system_results if r['correction_count'] > 0 and r['passed']
    ) / sum(1 for r in system_results if r['correction_count'] > 0)
    
    return {
        "baseline_pass1": baseline_pass1,
        "system_pass1": system_pass1,
        "improvement": system_pass1 - baseline_pass1,
        "baseline_tokens": baseline_avg_tokens,
        "system_tokens": system_avg_tokens,
        "token_savings_pct": token_savings,
        "baseline_time_sec": baseline_avg_time,
        "system_time_sec": system_avg_time,
        "avg_corrections": avg_corrections,
        "correction_success_rate": correction_success_rate
    }
```

```python
def test_significance(baseline_results, system_results):
    """Paired t-test for statistical significance."""
    
    baseline_scores = [1 if r['passed'] else 0 for r in baseline_results]
    system_scores = [1 if r['passed'] else 0 for r in system_results]
    
    t_stat, p_value = ttest_rel(baseline_scores, system_scores)
    
    return {
        "t_statistic": t_stat,
        "p_value": p_value,
        "significant": p_value < 0.05
    }
```

**Adaptive Threshold Tuning:**

```python
class AdaptiveValidator:
    def __init__(self):
        self.false_positive_history = []
        self.false_negative_history = []
        self.severity_threshold = 7  # Initial threshold
    
    def update_threshold(self, validation_outcome: dict):
        """Adjust severity threshold based on outcomes."""
        
        if validation_outcome['false_positive']:
            self.false_positive_history.append(validation_outcome['severity'])
        
        if validation_outcome['false_negative']:
            self.false_negative_history.append(validation_outcome['severity'])
        
        # If too many false positives, increase threshold
        if len(self.false_positive_history) > 10:
            avg_fp_severity = sum(self.false_positive_history[-10:]) / 10
            if avg_fp_severity < self.severity_threshold:
                self.severity_threshold += 0.5
        
        # If too many false negatives, decrease threshold
        if len(self.false_negative_history) > 10:
            avg_fn_severity = sum(self.false_negative_history[-10:]) / 10
            if avg_fn_severity > self.severity_threshold:
                self.severity_threshold -= 0.5
        
        # Clamp between 5-9
        self.severity_threshold = max(5, min(9, self.severity_threshold))
```

### 4.2 Limitations

**1. Single-Function Python Scope**

Current system limited to:
- Single-file Python tasks
- Standalone function implementations
- No multi-module dependencies

**Reason:** HumanEval dataset constraint + 1-month timeline

**2. Context Window Constraints**

For very long functions (>1000 lines):
- LLM context may become saturated
- Drift detection accuracy decreases with step count

**Mitigation:** Current scope (HumanEval max ~50 lines/function) avoids this

**3. Potential "Error Whack-a-Mole" Even with Partial Fixes**

Despite targeted regeneration:
- Fixing step 3 might introduce incompatibility with step 5
- Cascading corrections possible (though less frequent than full regen)

**Evidence:** Correction success rate 82% (18% still fail after correction)

**4. Deterministic Test Coverage**

System relies on provided test cases:
- Incomplete tests → drift may pass undetected
- No adversarial test generation

**5. LLM Critic Reliability**

LLM-as-judge approach has inherent limitations:
- Subject to same biases as generator
- May miss subtle semantic errors
- Temperature=0.2 reduces but doesn't eliminate non-determinism
